cmake_minimum_required(VERSION 3.26)



set(EDITOR_PROJECT_NAME editor)

project(${EDITOR_PROJECT_NAME} LANGUAGES CXX VERSION ${ENGINE_VERSION})



set(EDITOR_INCLUDES

		include/editor/controllers/registration_controller.hpp

		include/editor/Editor.hpp

)

set(EDITOR_SOURCES

		src/main.cpp

		src/editor/UI/registration_UI.cpp

		src/editor/controllers/registration_controller.cpp

		src/editor/Editor.cpp

)

set(EDITOR_ALL_FILES ${EDITOR_INCLUDES} ${EDITOR_SOURCES})



source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR} FILES ${EDITOR_ALL_FILES})



add_executable(${EDITOR_PROJECT_NAME}
		${EDITOR_ALL_FILES}		
)



target_include_directories(${EDITOR_PROJECT_NAME} PRIVATE include)
target_link_libraries(${EDITOR_PROJECT_NAME} PRIVATE uengine auth_client)



set_target_properties(${EDITOR_PROJECT_NAME} PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${ENGINE_BINARY_DIR})
